tradition_dragon_warriors = {
	category = combat
	
	layers = {
		0 = martial
		1 = western
		4 = war.dds
	}
	
	is_shown = {
		has_cultural_tradition = tradition_dragonlords
	}
	
	can_pick = {
		has_cultural_tradition = tradition_dragonlords
		NOT = { has_cultural_tradition = tradition_dragon_tamers }
	}
	
	parameters = {
		dragon_combat_bonus = yes
		aerial_assault_enabled = yes
		siege_dragon_bonus = yes
	}
	
	character_modifier = {
		prowess = 3
		martial = 1
		knight_effectiveness_mult = 0.15
		dread_gain_mult = 0.25
	}
	
	cost = {
		prestige = 2500
	}
	
	ai_will_do = {
		value = 100
		if = {
			limit = {
				culture_head = {
					has_trait = brave
				}
			}
			add = 100
		}
	}
}
